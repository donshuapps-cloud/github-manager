# Función para detectar información del proyecto
detect_project_info() {
    local name=""
    local version=""
    local description=""
    local author=""
    local license=""
    
    # Buscar en package.json
    if [ -f "package.json" ]; then
        name=$(grep -o '"name"[[:space:]]*:[[:space:]]*"[^"]*"' package.json | head -1 | cut -d'"' -f4)
        version=$(grep -o '"version"[[:space:]]*:[[:space:]]*"[^"]*"' package.json | head -1 | cut -d'"' -f4)
        description=$(grep -o '"description"[[:space:]]*:[[:space:]]*"[^"]*"' package.json | head -1 | cut -d'"' -f4)
        author=$(grep -o '"author"[[:space:]]*:[[:space:]]*"[^"]*"' package.json | head -1 | cut -d'"' -f4)
        license=$(grep -o '"license"[[:space:]]*:[[:space:]]*"[^"]*"' package.json | head -1 | cut -d'"' -f4)
    fi
    
    # Buscar en pyproject.toml
    if [ -f "pyproject.toml" ] && [ -z "$name" ]; then
        name=$(grep -o 'name[[:space:]]*=[[:space:]]*"[^"]*"' pyproject.toml | head -1 | cut -d'"' -f2)
        version=$(grep -o 'version[[:space:]]*=[[:space:]]*"[^"]*"' pyproject.toml | head -1 | cut -d'"' -f2)
        description=$(grep -o 'description[[:space:]]*=[[:space:]]*"[^"]*"' pyproject.toml | head -1 | cut -d'"' -f2)
    fi
    
    # Buscar en setup.py
    if [ -f "setup.py" ] && [ -z "$name" ]; then
        name=$(grep -o "name=['\"][^'\"]*['\"]" setup.py | head -1 | cut -d"'" -f2)
        version=$(grep -o "version=['\"][^'\"]*['\"]" setup.py | head -1 | cut -d"'" -f2)
    fi
    
    # Buscar en Cargo.toml (Rust)
    if [ -f "Cargo.toml" ] && [ -z "$name" ]; then
        name=$(grep -o 'name[[:space:]]*=[[:space:]]*"[^"]*"' Cargo.toml | head -1 | cut -d'"' -f2)
        version=$(grep -o 'version[[:space:]]*=[[:space:]]*"[^"]*"' Cargo.toml | head -1 | cut -d'"' -f2)
        description=$(grep -o 'description[[:space:]]*=[[:space:]]*"[^"]*"' Cargo.toml | head -1 | cut -d'"' -f2)
    fi
    
    # Si no hay nombre, usar el nombre del directorio
    if [ -z "$name" ]; then
        name=$(basename "$(pwd)")
    fi
    
    # Si no hay descripción, usar valor por defecto
    if [ -z "$description" ]; then
        description="Proyecto desarrollado con GitHub Repository Manager"
    fi
    
    echo "$name|$version|$description|$author|$license"
}

# Función para detectar tecnologías
detect_technologies() {
    local techs=()
    
    [ -f "package.json" ] && techs+=("Node.js")
    [ -f "requirements.txt" ] && techs+=("Python")
    [ -f "pyproject.toml" ] && techs+=("Python")
    [ -f "Cargo.toml" ] && techs+=("Rust")
    [ -f "go.mod" ] && techs+=("Go")
    [ -f "Gemfile" ] && techs+=("Ruby")
    [ -f "composer.json" ] && techs+=("PHP")
    [ -f "pom.xml" ] && techs+=("Java/Maven")
    [ -f "build.gradle" ] && techs+=("Java/Gradle")
    [ -f "Dockerfile" ] && techs+=("Docker")
    [ -f "docker-compose.yml" ] && techs+=("Docker Compose")
    [ -f ".github/workflows/*.yml" ] 2>/dev/null && techs+=("GitHub Actions")
    [ -f "Makefile" ] && techs+=("Make")
    [ -f "README.md" ] && techs+=("Documentation")
    
    # Detectar frameworks
    if [ -f "package.json" ]; then
        if grep -q '"react"' package.json 2>/dev/null; then
            techs+=("React")
        fi
        if grep -q '"vue"' package.json 2>/dev/null; then
            techs+=("Vue.js")
        fi
        if grep -q '"angular"' package.json 2>/dev/null; then
            techs+=("Angular")
        fi
        if grep -q '"next"' package.json 2>/dev/null; then
            techs+=("Next.js")
        fi
    fi
    
    if [ -z "${techs[*]}" ]; then
        techs+=("Shell Script")
    fi
    
    echo "${techs[@]}"
}

# Función para listar estructura de directorios
list_directory_structure() {
    local max_depth=3
    local max_items=20
    
    echo '```'
    tree -L "$max_depth" -I 'node_modules|.git|__pycache__|*.pyc' --dirsfirst 2>/dev/null || \
    find . -maxdepth "$max_depth" -not -path "*/\.*" -not -path "*/node_modules/*" -not -path "*/__pycache__/*" | head -"$max_items" | sort
    echo '```'
}

# Función para generar badges
generate_badges() {
    local repo_path="$1"
    local badges=""
    
    # Badge de versión
    badges+="![Version](https://img.shields.io/badge/version-${VERSION:-1.0.0}-blue.svg)\n"
    
    # Badge de licencia
    if [ -n "$LICENSE" ]; then
        badges+="![License](https://img.shields.io/badge/license-${LICENSE}-green.svg)\n"
    fi
    
    # Badge de lenguaje principal
    if [ -n "$GITHUB_TOKEN" ] && [ -n "$repo_path" ]; then
        local lang_response
        lang_response=$(curl -s -H "Authorization: token ${GITHUB_TOKEN}" \
                        "https://api.github.com/repos/${repo_path}/languages" 2>/dev/null)
        local main_lang
        main_lang=$(echo "$lang_response" | grep -o '"[^"]*":' | head -1 | tr -d '":')
        if [ -n "$main_lang" ]; then
            badges+="![Language](https://img.shields.io/badge/language-${main_lang}-orange.svg)\n"
        fi
    fi
    
    # Badge de GitHub Actions
    if [ -d ".github/workflows" ]; then
        badges+="![GitHub Actions](https://img.shields.io/badge/CI%2FCD-GitHub%20Actions-blue.svg)\n"
    fi
    
    # Badge de GitHub Pages
    if [ -n "$GITHUB_TOKEN" ] && [ -n "$repo_path" ]; then
        local pages_response
        pages_response=$(curl -s -H "Authorization: token ${GITHUB_TOKEN}" \
                        "https://api.github.com/repos/${repo_path}/pages" 2>/dev/null)
        if echo "$pages_response" | grep -q '"html_url"'; then
            badges+="![GitHub Pages](https://img.shields.io/badge/GitHub%20Pages-active-brightgreen.svg)\n"
        fi
    fi
    
    echo -e "$badges"
}

# Función para generar instalación y uso
generate_installation_usage() {
    local techs=("$@")
    local install_section=""
    local usage_section=""
    
    # Detectar tipo de proyecto para comandos específicos
    if [[ " ${techs[*]} " =~ "Node.js" ]]; then
        install_section="\`\`\`bash\nnpm install\n\`\`\`\n"
        usage_section="\`\`\`bash\nnpm start\n\`\`\`\n"
    elif [[ " ${techs[*]} " =~ "Python" ]]; then
        install_section="\`\`\`bash\npip install -r requirements.txt\n\`\`\`\n"
        usage_section="\`\`\`bash\npython main.py\n\`\`\`\n"
    elif [[ " ${techs[*]} " =~ "Rust" ]]; then
        install_section="\`\`\`bash\ncargo build --release\n\`\`\`\n"
        usage_section="\`\`\`bash\n./target/release/${PROJECT_NAME}\n\`\`\`\n"
    elif [[ " ${techs[*]} " =~ "Go" ]]; then
        install_section="\`\`\`bash\ngo mod download\ngo build\n\`\`\`\n"
        usage_section="\`\`\`bash\n./${PROJECT_NAME}\n\`\`\`\n"
    else
        install_section="\`\`\`bash\ngit clone https://github.com/${GITHUB_USER:-usuario}/${PROJECT_NAME}.git\ncd ${PROJECT_NAME}\n\`\`\`\n"
        usage_section="\`\`\`bash\n./github-manager.sh\n\`\`\`\n"
    fi
    
    echo "## Instalación\n\n$install_section"
    echo "## Uso\n\n$usage_section"
}

repo_docs() {

        log_info "Ejecuta la opción 1 o 2 primero para inicializar un repositorio"

    echo "      git add README.md LICENSE CONTRIBUTING.md CHANGELOG.md"
    echo "      git commit -m \"docs: generate project documentation\""
    echo "      git push"
    
    # Preguntar si desea commitear automáticamente
    echo ""
    read -p "¿Deseas hacer commit de la documentación? (s/n): " auto_commit
    if [[ "$auto_commit" =~ ^[Ss]$ ]]; then
        git add README.md 2>/dev/null || true
        [ -f "LICENSE" ] && git add LICENSE 2>/dev/null || true
        [ -f "CONTRIBUTING.md" ] && git add CONTRIBUTING.md 2>/dev/null || true
        [ -f "CHANGELOG.md" ] && git add CHANGELOG.md 2>/dev/null || true
        
        git commit -m "docs: generate project documentation" 2>/dev/null && log_success "Documentación commiteada" || log_info "No hay cambios para commitear"
        
        read -p "¿Deseas hacer push a GitHub? (s/n): " auto_push
        if [[ "$auto_push" =~ ^[Ss]$ ]]; then
            git push 2>/dev/null && log_success "Cambios subidos a GitHub" || log_warning "No se pudo hacer push automáticamente"
        fi
    fi
}
